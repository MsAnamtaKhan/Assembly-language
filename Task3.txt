.data 
char byte "c" 
msg1 byte "Both are equal",0
msg2 byte "entered char is smaller",0 
msg3 byte "entered char is bigger",0
.code 
main PROC
 call readchar  
 cmp al,char
 ;call writeint
 je L1

 jg L3  
 L1:   
 mov edx, offset msg1   
 call writestring

 L2:    mov edx, offset msg2   
 call writestring   
 jmp _exit 
 L3:   
 mov edx, offset msg3
 call writestring

 _exit:  
 call readint

main ENDP
